## STEP 7: X402 PAYMENT INTEGRATION

Implement the full x402 flow:

```typescript
// x402Client.ts
import { x402Fetch } from "x402-fetch";

export async function payWithX402(endpoint: string, wallet: any) {
  const res = await x402Fetch(endpoint, {
    method: "GET",
    wallet
  });
  return res.json();
}
```

```typescript  
// x402Middleware.ts — expose a paid endpoint example
import { paymentMiddleware } from "x402-express";
```

## STEP 8: AGENTIC TREASURY MANAGER

Build the AI cashflow agent loop:

```typescript
// scheduler.ts — runs every hour
const job = new Queue("cashflow-agent");

job.process(async () => {
  const vaults = await getAllActiveVaults();
  
  for (const vault of vaults) {
    // 1. Check collateral health
    const ratio = await checkCollateralRatio(vault.id);
    if (ratio < 1.5) await emitAlert(vault.id, "LOW_COLLATERAL");
    
    // 2. Check upcoming payments
    const upcoming = await getUpcomingPayments(vault.id);
    
    // 3. Check MUSD balance
    const balance = await getMUSDBalance(vault.id);
    
    // 4. Execute if due
    for (const payment of upcoming) {
      if (isDue(payment) && balance >= payment.amount) {
        if (payment.type === "x402") {
          await payWithX402(payment.endpoint, vault.wallet);
        } else {
          await contract.executePayment(payment.id);
        }
      }
    }
  }
});
```
